Console and variables :    
firstName = "Aaron";
var lastName = "ALAYO";
console.log("My first name is", firstName, "and my last name is", lastName); 
//shows in the console
My first name is Aaron and my last name is ALAYO


Numbers and Strings:

Add the year plus the number
The result should be 2020

var year = "2019";
var number = 1;
//first option
var newYear = parseInt(year) + number;
//second option
var newYear = Number(year) + number;
//thrid option
var newYear = +(year) + 1;

console.log(newYear);
//shows in the console
2020

var something = "123abc1";

console.log(parseInt(something));
//shows in the console
123
console.log(Number(something));
//shows in the console
123
       
Add numbers from string to float

var numberOne = "1.10";
var numberTwo = "2.30";

var newNumber = Number(numberOne) + Number(numberTwo);
console.log(newNumber);
//shows in the console
3.4

Add the numbers and the total with 2 decimals

var numberOne = "1.10";
var numberTwo = "2.30";

var addedNumbers = (parseFloat(numberOne) + parseFloat(numberTwo));
console.log(parseFloat(addedNumbers).toFixed(2));

//shows in the console
3.40
